可以直接传递json地址或json数据，可以定制单击事件及默认展开树节点 
$("#tree").treeview({
		url:'tree.json' //JSON地址
        ,data: json	    //json数据，以data数据为主，url为辅
		,click:function(i){alert($(i).attr('xtype'))} //点击事件,此处的i为a连接
		,openid:49 //默认打开ID
        ,field:{xtype:'type',href:'#g:id'} //为A标签构造参数，前者为A标签属性名，后者为json数据字段名
        									//特别注意href的写法，：前是固定值，后面是json数据字段名
        ,firstview:{field:'type',value:'router'}	//显示的顶级元素
        ,errorid:array //错误设备id 数组格式
        ,errorClass:'errorexp' //错误样式id
	});
